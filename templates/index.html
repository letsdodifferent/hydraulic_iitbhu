Geometry: Assi Ghat // =============================== var assiGhat = ee.Geometry.Point([83.0065, 25.2886]); // lon, lat Map.centerObject(assiGhat, 14); Map.addLayer(assiGhat, {color: 'red'}, 'Assi Ghat point'); // =============================== // 2. Choose dataset (Sentinel-2 SR) // =============================== // Change collection ID if you want Landsat etc. var collection = ee.ImageCollection('COPERNICUS/S2_SR') .filterBounds(assiGhat) .filterDate('2017-01-01', '2025-01-01') // <--- change dates if needed .filter(ee.Filter.lte('CLOUDY_PIXEL_PERCENTAGE', 20)); // cloud filter print('Number of images:', collection.size()); print('First image:', collection.first()); // =============================== // 3. Extract all band values at the point // =============================== var scale = 10; // Sentinel-2 native resolution for main bands var fc = collection.map(function (img) { // get all bands at the point var dict = img.reduceRegion({ reducer: ee.Reducer.first(), // take pixel value at the point geometry: assiGhat, scale: scale, bestEffort: true }); // turn dict into a Feature, add date & image id return ee.Feature(null, dict) .set('date', img.date().format('YYYY-MM-dd')) .set('image_id', img.id()); }); print('FeatureCollection to export:', fc.limit(5)); // =============================== // 4.
